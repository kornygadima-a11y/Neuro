    <script>
        const tg = window.Telegram.WebApp;
        let hero = 'female';
        const v = document.getElementById('bg-video');

        function setH(h) { 
            hero = h; 
            document.getElementById('s0').style.display='none'; 
            document.getElementById('s1').style.display='block'; 
            play(1, false); 
        }

        function toMenu() { 
            document.getElementById('s1').style.display='none'; 
            document.getElementById('s2').style.display='block'; 
            play(2, true); 
        }

        function act(n) { 
            document.getElementById('s2').style.display='none'; 
            play(n, false); 
            setTimeout(() => { play(3, true); setTimeout(() => { show(); }, 3000); }, 4000); 
        }

        function show() { 
            play(4, false); 
            document.getElementById('s3').style.display='block'; 
            document.getElementById('res').innerText = "Анализ завершен. Матрица подтверждает успех."; 
        }

        function play(n, loop) { 
            // ВНИМАНИЕ: Код теперь ищет твои файлы с двойным .mp4.mp4
            v.src = hero + '/' + n + '.mp4.mp4'; 
            v.loop = loop; 
            v.play(); 
        }

        function goSync() { act(3); }
        function back() { 
            document.getElementById('s3').style.display='none'; 
            document.getElementById('s2').style.display='block'; 
            play(2, true); 
        }
